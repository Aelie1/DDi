RequireVersions "", "", "", "303"


; -----ANIMATIONS-----

abcDefault = False
abcMaikCG = False
abcFeuertin = False

SelectOne "Which style of armbinder and yoke animations do you prefer?", \
	"Cedec0 (default)", "Animations by cedec0. Abundant upper body movement, omnidirectional running and walking.", "fomod/images/01_cedec0.png", \
	"MaikCG", "Animations by MaikCG. More upright posture, omnidirectional running, walking and sneaking.", "fomod/images/01_maikcg.png", \
	"Feuertin", "Animations by Feuertin. Feminine but subtle, suitable for both genders, foward-facing running and walking.", "fomod/images/01_feuertin.png"

	Case "Cede0 (default)"
		abcDefault = True
		Break
	Case "MaikCG"
		abcMaikCG = True
		Break
	Case "Feuertin"
		abcFeuertin = True
		Break
EndSelect

hbcDefault = False
hbcBunny = False

SelectOne "Which style of hobble skirt animations do you prefer?", \
	"Walking (default)", "Animations by Feuertin. Slowly pushing feet forward one at a time. Optimised for low movement speed.", "fomod/images/02_walk.png", \
	"Bunnyhopping", "Animations by Feuertin. Hop, hop, hop! Optimised for high movement speed.", "fomod/images/02_bunny.png", \

	Case "Walking (default)"
		hbcDefault = True
		Break
	Case "Bunnyhopping"
		hbcBunny = True
		Break
EndSelect

; -----INSTALL-----

DeSelectAll
SelectSubPackage "00 Core"
If abcMaikCG
	SelectSubPackage "01 MaikCG"
Elif abcFeuertin
	SelectSubPackage "01 Feuertin"
EndIf
If hbcBunny
	SelectSubPackage "02 Bunnyhop"
EndIf
